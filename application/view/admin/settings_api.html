    XML-RPC PHP 7 Code example

    You can get the API key from the client side: My Account > Api Key.

    <?php
    // sheel_xmlrpc_itemupdate.php
    // This code uses the method item.update as an example for updating item details remotely.
    $cookiefile = "cookies.txt";
    if (!file_exists($cookiefile)) {$fh = fopen($cookiefile, "w");fwrite($fh, "");fclose($fh);}
    $params = array(
        'itemid' => 93950958,
        'sku' => '',
        'price' => '25.50',
        'qty' => '22',
        'username' => 'xxxxxx',
        'password' => 'xxxxxx',
        'apikey' => 'xxxxxx',
        'csrftoken' => ''
    );
    $request = xmlrpc_encode_request('item.update', $params);
    $ch = curl_init();
    curl_setopt($ch, CURLOPT_URL, '{https_server}rpc/');
    curl_setopt($ch, CURLOPT_USERAGENT, 'Mozilla/5.0 (Windows; U; Windows NT 5.1; en-US; rv:1.8.1.13) Gecko/20080311 Firefox/2.0.0.13');
    curl_setopt($ch, CURLOPT_POSTFIELDS, $request);
    curl_setopt($ch, CURLOPT_RETURNTRANSFER, 1);
    curl_setopt($ch, CURLOPT_TIMEOUT, 30);
    curl_setopt($ch, CURLOPT_HEADER, 0);
    curl_setopt($ch, CURLOPT_COOKIEFILE, $cookiefile);
    curl_setopt($ch, CURLOPT_COOKIEJAR, $cookiefile);
    $results = curl_exec($ch);
    $results = xmlrpc_decode($results);
    curl_close($ch);
    print_r($results);
    ?>
